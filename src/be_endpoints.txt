### alle User bekommen 
curl -X 'GET' \
  'http://127.0.0.1:8000/api/v1/users/?skip=0&limit=100' \
  -H 'accept: application/json'

###get user by id
curl -X 'GET' \
  'http://127.0.0.1:8000/api/v1/users/4' \
  -H 'accept: application/json'

###update user

 curl -X 'PUT' \
  'http://127.0.0.1:8000/api/v1/users/4' \
  -H 'accept: application/json' \
  -H 'Content-Type: application/json' \
  -d '{
  "email": "user@example.com",
  "username": "string",
  "password": "string",
  "birthdate": "2024-06-04",
  "is_active": true
}'

###delete user 

curl -X 'DELETE' \
  'http://127.0.0.1:8000/api/v1/users/4' \
  -H 'accept: application/json'
